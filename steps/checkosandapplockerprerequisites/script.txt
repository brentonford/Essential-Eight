$os = Get-CimInstance -ClassName Win32_OperatingSystem
Write-Output "OS Caption : $($os.Caption)"
Write-Output "OS Version : $($os.Version)"
Write-Output "OS Build : $($os.BuildNumber)"

$appLockerModule = Get-Module -ListAvailable -Name AppLocker
if ($null -eq $appLockerModule) {
    Write-Error "AppLocker PowerShell module is not available on this system."
    exit 1
}

Write-Output "AppLocker module : Available"
Write-Output "Prerequisites check passed."
